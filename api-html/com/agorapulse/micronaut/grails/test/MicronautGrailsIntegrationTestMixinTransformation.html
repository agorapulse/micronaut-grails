102     static final ClassNode MY_TYPE = new ClassNode(MicronautGrailsIntegration)
103     public static final ClassNode CONTEXT_CONFIG_ANNOTATION = ClassHelper.make(ContextConfiguration)
104     public static final ClassNode GRAILS_APPLICATION_CONTEXT_LOADER = ClassHelper.make(MicronautGrailsApplicationContextLoader)
105     public static final ClassNode INTEGRATION_TEST_CLASS_NODE = ClassHelper.make(SpringBootTest)
106     public static final ClassNode SPRING_APPLICATION_CONFIGURATION_CLASS_NODE = ClassHelper.make(GrailsTestConfiguration)
107     public static final ClassNode RUN_WITH_ANNOTATION_NODE = ClassHelper.make(RunWith)
108     public static final ClassNode SPRING_JUNIT4_CLASS_RUNNER = ClassHelper.make(GrailsJunit4ClassRunner)
109     public static final String SPEC_CLASS = "spock.lang.Specification"

111     @Override
112     void visit(ASTNode[] astNodes, SourceUnit source) {
113         if (!(astNodes[0] instanceof AnnotationNode) || !(astNodes[1] instanceof AnnotatedNode)) {
114             throw new RuntimeException("Internal error: wrong types: ${astNodes[0].getClass()} / ${astNodes[1].getClass()}")
115         }
116 
117         AnnotatedNode parent = (AnnotatedNode) astNodes[1]
118         AnnotationNode annotationNode = (AnnotationNode) astNodes[0]
119         if (!MY_TYPE.equals(annotationNode.classNode) || !(parent instanceof ClassNode)) {
120             return
121         }
122 
123         ClassExpression applicationClassExpression = (ClassExpression)annotationNode.getMember('applicationClass')
124 
125         final applicationClassNode
126         if(applicationClassExpression) {
127             applicationClassNode = applicationClassExpression.getType()
128             if(!applicationClassNode.isDerivedFrom(ClassHelper.make(MicronautGrailsAutoConfiguration))) {
129                 GrailsASTUtils.error(source, applicationClassExpression, "Invalid applicationClass attribute value [${applicationClassNode.getName()}].  The applicationClass attribute must specify a class which extends grails.boot.config.GrailsAutoConfiguration.", true)
130             }
131         } else {
132             String mainClass = MainClassFinder.searchMainClass(source.source.URI)
133             if(mainClass) {
134                 applicationClassNode = ClassHelper.make(mainClass)
135             } else {
136                 applicationClassNode = null
137             }
138         }
139 
140         if(applicationClassNode) {
141             ClassNode classNode = (ClassNode) parent
142 
143             weaveIntegrationTestMixin(classNode, applicationClassNode)
144 
145         }
146 
147     }
148 
149     public void weaveIntegrationTestMixin(ClassNode classNode, ClassNode applicationClassNode) {
150         if(applicationClassNode == null) return
151 
152 
153         enableAutowireByName(classNode)
154 
155         if (GrailsASTUtils.isSubclassOf(classNode, SPEC_CLASS)) {
156             // first add context configuration
157             // Example: @ContextConfiguration(loader = GrailsApplicationContextLoader, classes = Application)
158             def contextConfigAnn = new AnnotationNode(CONTEXT_CONFIG_ANNOTATION)
159             contextConfigAnn.addMember("loader", new ClassExpression(GRAILS_APPLICATION_CONTEXT_LOADER))
160             contextConfigAnn.addMember("classes", new ClassExpression(applicationClassNode))
161             classNode.addAnnotation(contextConfigAnn)
162 
163             enhanceGebSpecWithPort(classNode)
164 
165         } else {
166             // Must be a JUnit 4 test so add JUnit spring annotations
167             // @RunWith(SpringJUnit4ClassRunner)
168             def runWithAnnotation = new AnnotationNode(RUN_WITH_ANNOTATION_NODE)
169             runWithAnnotation.addMember("value", new ClassExpression(SPRING_JUNIT4_CLASS_RUNNER))
170             classNode.addAnnotation(runWithAnnotation)
171 
172             // @SpringApplicationConfiguration(classes = Application)
173             def contextConfigAnn = new AnnotationNode(SPRING_APPLICATION_CONFIGURATION_CLASS_NODE)
174             contextConfigAnn.addMember("classes", new ClassExpression(applicationClassNode))
175             classNode.addAnnotation(contextConfigAnn)
176         }
177 
178         // now add integration test annotations
179         // @SpringBootTest
180         def servletApi = null
181         try {
182             servletApi = Class.forName("javax.servlet.ServletContext", false, getClass().classLoader)
183         }
184         catch(Exception e) {
185             // ignore
186         }
187 
188 
189         if (servletApi != null) {
190 
191             if( GrailsASTUtils.findAnnotation(classNode, SpringBootTest) == null) {
192                 AnnotationNode webIntegrationTestAnnotation = GrailsASTUtils.addAnnotationOrGetExisting(classNode, SpringBootTest)
193                 webIntegrationTestAnnotation.addMember("webEnvironment", propX(classX(SpringBootTest.WebEnvironment), "RANDOM_PORT"))
194                 if(classNode.getProperty("serverPort") == null) {
195 
196                     def serverPortField = new FieldNode("serverPort", Modifier.PROTECTED, ClassHelper.Integer_TYPE, classNode, new ConstantExpression(8080))
197                     def valueAnnotation = new AnnotationNode(ClassHelper.make(Value))
198                     valueAnnotation.setMember("value", new ConstantExpression('${local.server.port}'))
199                     serverPortField.addAnnotation(valueAnnotation)
200 
201                     classNode.addProperty(new PropertyNode(serverPortField, Modifier.PUBLIC, null, null ))
202                 }
203             }
204         } else {
205             classNode.addAnnotation(new AnnotationNode(INTEGRATION_TEST_CLASS_NODE))
206         }
207     }
208 
209     protected void enableAutowireByName(ClassNode classNode) {
210         classNode.addInterface(ClassHelper.make(ApplicationContextAware))
211 
212         def body = new BlockStatement()
213 
214         def ctxClass = ClassHelper.make(ApplicationContext)
215         def p = new Parameter(ctxClass, "ctx")
216 
217         def getBeanFactoryMethodCall = new MethodCallExpression(new VariableExpression(p), "getAutowireCapableBeanFactory", GrailsASTUtils.ZERO_ARGUMENTS)
218 
219         def args = new ArgumentListExpression(new VariableExpression("this"), new ConstantExpression(AutowireCapableBeanFactory.AUTOWIRE_BY_NAME), new ConstantExpression(Boolean.FALSE))
220         def autoMethodCall = new MethodCallExpression(getBeanFactoryMethodCall, "autowireBeanProperties", args)
221         body.addStatement(new ExpressionStatement(autoMethodCall))
222         classNode.addMethod("setApplicationContext", Modifier.PUBLIC, ClassHelper.VOID_TYPE, [p] as Parameter[], null, body)
223     }
224 
225     protected void enhanceGebSpecWithPort(ClassNode classNode) {
226         if (GrailsASTUtils.isSubclassOf(classNode, "geb.spock.GebSpec")) {
227             def contextPathParameter = new Parameter(ClassHelper.make(String), 'serverContextPath')
228             def cpValueAnnotation = new AnnotationNode(ClassHelper.make(Value))
229             cpValueAnnotation.setMember('value', new ConstantExpression('${server.servlet.context-path:/}'))
230             contextPathParameter.addAnnotation(cpValueAnnotation)
231 
232             def serverPortParameter = new Parameter(ClassHelper.make(Integer.TYPE), 'serverPort')
233             def spValueAnnotation = new AnnotationNode(ClassHelper.make(Value))
234             spValueAnnotation.setMember('value', new ConstantExpression('${local.server.port}'))
235             serverPortParameter.addAnnotation(spValueAnnotation)
236 
237             Expression urlExpression = new GStringExpression('http://localhost:${serverPort}${serverContextPath}', [new ConstantExpression('http://localhost:'), new ConstantExpression(""), new ConstantExpression("")], [new VariableExpression('serverPort'), new VariableExpression('serverContextPath')] as List<Expression>)
238 
239             Expression baseUrlVariableExpression = new VariableExpression('$baseUrl', ClassHelper.make(String))
240             Expression declareBaseUrlExpression = new DeclarationExpression(baseUrlVariableExpression, Token.newSymbol(Types.EQUALS, 0, 0), new MethodCallExpression(urlExpression, 'toString', new ArgumentListExpression()))
241 
242             Expression constantSlashExpression = new ConstantExpression('/')
243             Expression endsWithMethodCall = new MethodCallExpression(baseUrlVariableExpression, 'endsWith', new ArgumentListExpression(constantSlashExpression))
244             Expression appendSlashExpression = new BinaryExpression(baseUrlVariableExpression, Token.newSymbol(Types.PLUS_EQUAL, 0, 0), constantSlashExpression)
245             Statement ifUrlEndsWithSlashStatement = new IfStatement(new BooleanExpression(endsWithMethodCall), new EmptyStatement(), new ExpressionStatement(appendSlashExpression))
246             def methodBody = new BlockStatement()
247             methodBody.addStatement(new ExpressionStatement(declareBaseUrlExpression))
248             methodBody.addStatement(ifUrlEndsWithSlashStatement)
249             def systemClassExpression = new ClassExpression(ClassHelper.make(System))
250             def args = new ArgumentListExpression()
251             args.addExpression(new ConstantExpression("geb.build.baseUrl"))
252             args.addExpression(baseUrlVariableExpression)
253             methodBody.addStatement(new ExpressionStatement(new MethodCallExpression(systemClassExpression, "setProperty", args)))
254             def method = new MethodNode("configureGebBaseUrl", Modifier.PUBLIC, ClassHelper.VOID_TYPE, [contextPathParameter, serverPortParameter] as Parameter[], null, methodBody)
255             method.addAnnotation(new AnnotationNode(ClassHelper.make(Autowired)))
256             classNode.addMethod(method)
257         }
258     }
