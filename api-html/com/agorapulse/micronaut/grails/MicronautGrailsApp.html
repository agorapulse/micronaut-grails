057     public enum Compatibility {
058         /**
059          * Runs the application in the legacy mode.
060          *
061          * <ul>
062          *     <li>Two Micronaut contexts are created - one by Grails App and another one by Micronaut Grails for beans imported using {@link MicronautBeanImporter}</li>
063          *     <li>Micronaut beans declared using {@link MicronautBeanImporter} can be injected by name without <code>@Inject</code> annotation</li>
064          *     <li>Property prefixes are stripped by beans imported using {@link MicronautBeanImporter} according to existing {@link PropertyTranslatingCustomizer} beans</li>
065          *     <li>Properties are ignored by beans imported using {@link MicronautBeanImporter} according to existing {@link PropertyTranslatingCustomizer} beans</li>
066          * </ul>
067          */
068         @Deprecated
069         LEGACY(ENVIRONMENT_LEGACY),
070 
071         /**
072          * Runs the application in the bridge mode.
073          *
074          * <ul>
075          *     <li>Only one Micronaut context is created - the one by Grails App</li>
076          *     <li>Micronaut beans declared using {@link MicronautBeanImporter} can be injected by name without <code>@Inject</code> annotation</li>
077          *     <li>Property prefixes are <em>no longer stripped</em> by beans imported using {@link MicronautBeanImporter} according to existing {@link PropertyTranslatingCustomizer} beans</li>
078          *     <li>Properties are <em>no longer ignored</em> by beans imported using {@link MicronautBeanImporter} according to existing {@link PropertyTranslatingCustomizer} beans</li>
079          * </ul>
080          */
081         BRIDGE(ENVIRONMENT_BRIDGE),
082 
083         /**
084          * Runs the application in the bridge mode.
085          *
086          * <ul>
087          *     <li>Only one Micronaut context is created - the one by Grails App</li>
088          *     <li>All Micronaut beans must use <code>@Inject</code> annotation to be injected</li>
089          *     <li>Property prefixes are <em>no longer stripped</em> by beans imported using {@link MicronautBeanImporter} according to existing {@link PropertyTranslatingCustomizer} beans</li>
090          *     <li>Properties are <em>no longer ignored</em> by beans imported using {@link MicronautBeanImporter} according to existing {@link PropertyTranslatingCustomizer} beans</li>
091          * </ul>
092          */
093         STRICT(ENVIRONMENT_STRICT);

282     /**
283      * Strategy method used to create the {@link org.springframework.context.ApplicationContext}. By default this
284      * method will respect any explicitly set application context or application context
285      * class before falling back to a suitable default.
286      *
287      * @return the application context (not yet refreshed)
288      * @see #setApplicationContextClass(Class)
289      */
290     protected ConfigurableApplicationContext createSpringApplicationContext() {
291         Class<?> contextClass;
292         try {
293             switch (this.getWebApplicationType()) {
294                 case SERVLET:
295                     contextClass = Class.forName(DEFAULT_SERVLET_WEB_CONTEXT_CLASS);
296                     break;
297                 case REACTIVE:
298                     contextClass = Class.forName(DEFAULT_REACTIVE_WEB_CONTEXT_CLASS);
299                     break;
300                 default:
301                     contextClass = Class.forName(DEFAULT_CONTEXT_CLASS);
302             }
303         } catch (ClassNotFoundException ex) {
304             throw new IllegalStateException("Unable create a default ApplicationContext, please specify an ApplicationContextClass", ex);
305         }
306         return (ConfigurableApplicationContext) BeanUtils.instantiateClass(contextClass);
307     }

325     @Override
326     protected ConfigurableApplicationContext createApplicationContext() {
327         setAllowBeanDefinitionOverriding(true);
328 
329         ConfigurableApplicationContext applicationContext = createSpringApplicationContext();
330 
331         long now = System.currentTimeMillis();
332 
333         ClassLoader applicationClassLoader = GrailsApp.class.getClassLoader();
334         ApplicationContextConfiguration micronautConfiguration = new MicronautGrailsAppContextConfiguration(applicationClassLoader);
335 
336         List<Class<?>> beanExcludes = new ArrayList<>();
337         beanExcludes.add(ConversionService.class);
338         beanExcludes.add(org.springframework.core.env.Environment.class);
339         beanExcludes.add(PropertyResolver.class);
340         beanExcludes.add(ConfigurableEnvironment.class);
341         ClassUtils.forName("com.fasterxml.jackson.databind.ObjectMapper", getClassLoader()).ifPresent(beanExcludes::add);
342         ApplicationContext micronautContext = new MicronautGrailsAppContext(micronautConfiguration);
343 
344         micronautContext.getEnvironment().addPropertySource("grails-config", Collections.singletonMap(MicronautBeanFactoryConfiguration.PREFIX + ".bean-excludes", beanExcludes));
345         micronautContext.registerSingleton(Configuration.class, configuration);
346 
347         micronautContext.start();
348 
349         ConfigurableApplicationContext parentContext = micronautContext.getBean(ConfigurableApplicationContext.class);
350         applicationContext.setParent(parentContext);
351         applicationContext.addApplicationListener(new MicronautShutdownListener(micronautContext));
352 
353         LOGGER.info("Started Micronaut Parent Application Context in " + (System.currentTimeMillis() - now) + " ms");
354 
355         if (isEnableBeanCreationProfiler()) {
356             BeanCreationProfilingPostProcessor processor = new BeanCreationProfilingPostProcessor();
357             applicationContext.getBeanFactory().addBeanPostProcessor(processor);
358             applicationContext.addApplicationListener(processor);
359         }
360         return applicationContext;
361     }
