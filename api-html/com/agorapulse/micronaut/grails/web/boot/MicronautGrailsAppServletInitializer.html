36 /**
37  * Ensure a {@link com.agorapulse.micronaut.grails.MicronautGrailsApp} in constructed during servlet initialization.
38  *
39  * Inspired by {@code GrailsAppServletInitializer}
40  *
41  */
42 public abstract class MicronautGrailsAppServletInitializer extends SpringBootServletInitializer {
43 
44     /*
45      * TODO: Replace with createSpringApplicationBuilder() override when upgrading to Spring Boot 1.3 M4 and delete this method
46      */
47 
48     protected WebApplicationContext createRootApplicationContext(ServletContext servletContext) {
49         SpringApplicationBuilder builder = new MicronautGrailsAppBuilder();
50         builder.main(getClass());
51         ApplicationContext parent = getExistingRootWebApplicationContext(servletContext);
52         if (parent != null) {
53             this.logger.info("Root context already created (using as parent).");
54             servletContext.setAttribute(WebApplicationContext.ROOT_WEB_APPLICATION_CONTEXT_ATTRIBUTE, null);
55             builder.initializers(new ParentContextApplicationContextInitializer(parent));
56         }
57         builder.initializers(new ServletContextApplicationContextInitializer(servletContext));
58         builder.application().setApplicationContextClass(AnnotationConfigServletWebServerApplicationContext.class);
59         builder = configure(builder);
60         SpringApplication application = builder.build();
61         if (application.getAllSources().isEmpty() && AnnotationUtils.findAnnotation(getClass(), Configuration.class) != null) {
62             application.getSources().add(getClass().getName());
63         }
64         Assert.state(
65             application.getAllSources().size() > 0,
66             "No SpringApplication sources have been defined. Either override the configure method or add an @Configuration annotation"
67         );
68         // Ensure error pages are registered
69         application.getSources().add(ErrorPageFilter.class.getName());
70         return run(application);
71     }
72 
73     private ApplicationContext getExistingRootWebApplicationContext(ServletContext servletContext) {
74         Object context = servletContext.getAttribute(WebApplicationContext.ROOT_WEB_APPLICATION_CONTEXT_ATTRIBUTE);
75         if (context instanceof ApplicationContext) {
76             return (ApplicationContext) context;
77         }
78         return null;
79     }
