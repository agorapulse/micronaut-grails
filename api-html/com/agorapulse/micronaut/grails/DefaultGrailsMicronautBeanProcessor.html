018 package com.agorapulse.micronaut.grails;
019 
020 import io.micronaut.context.Qualifier;
021 import io.micronaut.inject.BeanDefinition;
022 import org.slf4j.Logger;
023 import org.slf4j.LoggerFactory;
024 import org.springframework.beans.BeansException;
025 import org.springframework.beans.factory.DisposableBean;
026 import org.springframework.beans.factory.config.BeanFactoryPostProcessor;
027 import org.springframework.beans.factory.config.ConfigurableListableBeanFactory;
028 import org.springframework.beans.factory.support.BeanDefinitionBuilder;
029 import org.springframework.beans.factory.support.DefaultListableBeanFactory;
030 import org.springframework.context.ApplicationContext;
031 import org.springframework.context.ApplicationContextAware;
032 
033 import javax.annotation.Nonnull;
034 import java.net.URLClassLoader;
035 import java.util.Arrays;
036 import java.util.Collection;
037 import java.util.Map;
038 import java.util.Optional;
039 import java.util.stream.Collectors;
040 
041 /**
042  * Adds Micronaut beans to a Grails' Spring application context.  This processor will
043  * find all of the Micronaut beans of the specified types and add them as beans to the Spring application context.
044  *
045  * No property translation will happen.
046  * <p>
047  *
048  * @since 2.0.3
049  */
050 public class DefaultGrailsMicronautBeanProcessor implements BeanFactoryPostProcessor, DisposableBean, ApplicationContextAware {
051 
052     static final Logger LOGGER = LoggerFactory.getLogger(DefaultGrailsMicronautBeanProcessor.class);
053 
054     private static final String MICRONAUT_BEAN_TYPE_PROPERTY_NAME = "micronautBeanType";
055     private static final String MICRONAUT_CONTEXT_PROPERTY_NAME = "micronautContext";
056     private static final String MICRONAUT_QUALIFIER_PROPERTY_NAME = "micronautQualifier";
057     private static final String MICRONAUT_SINGLETON_PROPERTY_NAME = "micronautSingleton";
058 
059     private io.micronaut.context.ApplicationContext micronautContext;
060     private ApplicationContext springContext;
061     private final Map<String, TypeAndQualifier<?>> micronautBeanQualifiers;
062 
063     /**
064      * @param qualifiers the names and qualifiers of the Micronaut beans which should be added to the
065      *                   Spring application context.
066      */
067     DefaultGrailsMicronautBeanProcessor(Map<String, TypeAndQualifier<?>> qualifiers) {
068         this.micronautBeanQualifiers = qualifiers;
069     }
070 
071     @Override
072     public void postProcessBeanFactory(ConfigurableListableBeanFactory beanFactory) throws BeansException {
073         micronautContext = initializeMicronautContext();
074 
075         NoClassDefFoundError noClassDefFoundError = null;
076 
077         for (Map.Entry<String, TypeAndQualifier<?>> entry : micronautBeanQualifiers.entrySet()) {
078             String name = entry.getKey();
079             Class type = entry.getValue().getType();
080             Qualifier micronautBeanQualifier = entry.getValue().getQualifier();
081             try {
082                 Collection<BeanDefinition<?>> beanDefinitions = type == null
083                     ? micronautContext.getBeanDefinitions(micronautBeanQualifier)
084                     : micronautContext.getBeanDefinitions(type, micronautBeanQualifier);
085 
086                 if (beanDefinitions.size() > 1) {
087                     throw new IllegalArgumentException("There is too many candidates of type '" + type + "' for qualifier '" + micronautBeanQualifier + "'! Candidates: " + beanDefinitions);
088                 }
089 
090                 Optional<BeanDefinition<?>> firstBean = beanDefinitions.stream().findFirst();
091                 BeanDefinition<?> definition = firstBean.orElseThrow(()-> {
092                         String message = "There is no candidate for type " + type + " and qualifier " + micronautBeanQualifier + "\n"
093                             + "Known beans:\n" + micronautContext.getAllBeanDefinitions().stream().map(d -> d.getBeanType().getName() + " " + d.getName()).collect(Collectors.joining("\n  "));
094 
095                         return new IllegalArgumentException(message);
096                 });
097 
098                 final BeanDefinitionBuilder beanDefinitionBuilder = BeanDefinitionBuilder
099                     .rootBeanDefinition(GrailsMicronautBeanFactory.class);
100                 beanDefinitionBuilder.addPropertyValue(MICRONAUT_BEAN_TYPE_PROPERTY_NAME, type == null ? definition.getBeanType() : type);
101                 beanDefinitionBuilder.addPropertyValue(MICRONAUT_QUALIFIER_PROPERTY_NAME, micronautBeanQualifier);
102                 beanDefinitionBuilder.addPropertyValue(MICRONAUT_CONTEXT_PROPERTY_NAME, micronautContext);
103                 beanDefinitionBuilder.addPropertyValue(MICRONAUT_SINGLETON_PROPERTY_NAME, definition.isSingleton());
104 
105                 ((DefaultListableBeanFactory) beanFactory).registerBeanDefinition(name, beanDefinitionBuilder.getBeanDefinition());
106             } catch (NoClassDefFoundError error) {
107                 LOGGER.error("Exception loading class for qualifier {}. Bean {} will not be available in the runtime", micronautBeanQualifier, name);
108                 LOGGER.error("Current class loader: {}", printClassLoader(getClass().getClassLoader()));
109                 LOGGER.error("Parent class loader: {}",  printClassLoader(getClass().getClassLoader().getParent()));
110                 LOGGER.error("Current class path: {}", System.getProperty("java.class.path"));
111                 noClassDefFoundError = error;
112             }
113         }
114 
115         if (noClassDefFoundError == null) {
116             if (LOGGER.isInfoEnabled()) {
117                 LOGGER.info("Successfully added following beans to the spring contest {} ", micronautBeanQualifiers);
118                 LOGGER.debug("Current class loader: {}", printClassLoader(getClass().getClassLoader()));
119                 LOGGER.debug("Parent class loader: {}",  printClassLoader(getClass().getClassLoader().getParent()));
120                 LOGGER.trace("Current class path: {}", System.getProperty("java.class.path"));
121             }
122             return;
123         }
124 
125         throw noClassDefFoundError;
126     }
127 
128     protected io.micronaut.context.ApplicationContext initializeMicronautContext() {
129         return springContext.getBean(MicronautContextHolder.class).getContext();
130     }
131 
132     private static String printClassLoader(ClassLoader classLoader) {
133         if (classLoader instanceof URLClassLoader) {
134             return "URLClassLoader for URLS:" + Arrays.toString(((URLClassLoader) classLoader).getURLs());
135         }
136         if (classLoader == null) {
137             return null;
138         }
139         return classLoader.toString();
140     }
141 
142     @Override
143     public void destroy() {
144         if (micronautContext != null) {
145             micronautContext.close();
146         }
147     }
148 
149     @Override
150     public void setApplicationContext(@Nonnull ApplicationContext applicationContext) throws BeansException {
151         this.springContext = applicationContext;
152     }
153 }
