18  package com.agorapulse.micronaut.grails;
19  
20  import io.micronaut.context.Qualifier;
21  import io.micronaut.core.naming.NameUtils;
22  import io.micronaut.inject.qualifiers.Qualifiers;
23  
24  import java.lang.annotation.Annotation;
25  import java.util.*;
26  
27  /**
28   * Defineds which Micronaut beans should be added into a Grails' Spring application context. The processor will then
29   * find all of the Micronaut beans of the specified types and add them as beans to the Spring application context.
30   * <p>
31   * <p>
32   * It can also manage beans to
33   *
34   * @since 1.2.3.1
35   */
36  public class MicronautBeanImporter {
37  
38      public static MicronautBeanImporter create() {
39          return new MicronautBeanImporter();
40      }
41  
42      private final Map<String, TypeAndQualifier<?>> micronautBeanQualifiers = new LinkedHashMap<>();
43      private final List<PropertyTranslatingCustomizer> customizers = new ArrayList<>();
44      private final List<String> expectedMapProperties = new ArrayList<>();
45  
46      protected MicronautBeanImporter() {}
47  
48      public MicronautBeanImporter addByType(Class<?> type) {
49          return addByType(NameUtils.decapitalize(type.getSimpleName()), type);
50      }
51  
52      public MicronautBeanImporter addByType(String grailsBeanName, Class<?> type, Class<?>... types) {
53          if (types.length == 0) {
54              micronautBeanQualifiers.put(grailsBeanName, new TypeAndQualifier<>(type, null));
55          } else {
56              micronautBeanQualifiers.put(grailsBeanName, new TypeAndQualifier<>(type, Qualifiers.byType(types)));
57          }
58          return this;
59      }
60  
61      public MicronautBeanImporter addByStereotype(String grailsBeanName, Class<?> beanType, Class<? extends Annotation> type) {
62          return addByQualifiers(grailsBeanName, beanType, Qualifiers.byStereotype(type));
63      }
64  
65      /**
66       * @deprecated use {@link #addByStereotype(String, Class, Class)} instead
67       */
68      public MicronautBeanImporter addByStereotype(String grailsBeanName, Class<? extends Annotation> type) {
69          return addByQualifiers(grailsBeanName, Qualifiers.byStereotype(type));
70      }
71  
72      /**
73       * @deprecated use {@link #addByName(String, Class)} instead
74       */
75      public MicronautBeanImporter addByName(String name) {
76          return addByName(name, name);
77      }
78  
79      public MicronautBeanImporter addByName(String name, Class<?> beanType) {
80          return addByName(name, beanType, name);
81      }
82  
83      /**
84       * @deprecated use {@link #addByName(String, Class, String)} instead
85       */
86      public MicronautBeanImporter addByName(String grailsBeanName, String micronautName) {
87          micronautBeanQualifiers.put(grailsBeanName, new TypeAndQualifier<>(null, Qualifiers.byName(micronautName)));
88          return this;
89      }
90  
91      public MicronautBeanImporter addByName(String grailsBeanName, Class<?> beanType, String micronautName) {
92          micronautBeanQualifiers.put(grailsBeanName, new TypeAndQualifier<>(beanType, Qualifiers.byName(micronautName)));
93          return this;
94      }
95  
96      @SafeVarargs
97      public final <T> MicronautBeanImporter addByQualifiers(String grailsBeanName, Class<T> beanType, Qualifier<T>... qualifiers) {
98          micronautBeanQualifiers.put(grailsBeanName, new TypeAndQualifier<>(beanType, Qualifiers.byQualifiers(qualifiers)));
99          return this;
100     }
101 
102     /**
103      * @deprecated use {@link #addByQualifiers(String, Class, Qualifier[])} instead
104      */
105     @SafeVarargs
106     public final <T> MicronautBeanImporter addByQualifiers(String grailsBeanName, Qualifier<T>... qualifiers) {
107         return addByQualifiers(grailsBeanName, (Class<T>) null, Qualifiers.byQualifiers(qualifiers));
108     }
109 
110     public MicronautBeanImporter customize(PropertyTranslatingCustomizer customizer) {
111         this.customizers.add(customizer);
112         return this;
113     }
114 
115     public MicronautBeanImporter customize(PropertyTranslatingCustomizer.Builder customizer) {
116         return customize(customizer.build());
117     }
118 
119     public MicronautBeanImporter createMapForPropertiesStarting(String prefix) {
120         expectedMapProperties.add(prefix);
121         return this;
122     }
123 
124     public List<String> getExpectedMapProperties() {
125         return Collections.unmodifiableList(expectedMapProperties);
126     }
127 
128     public Map<String, TypeAndQualifier<?>> getMicronautBeanQualifiers() {
129         return Collections.unmodifiableMap(micronautBeanQualifiers);
130     }
131 
132     public List<PropertyTranslatingCustomizer> getCustomizers() {
133         return Collections.unmodifiableList(customizers);
134     }
135 
136     /**
137      * @return new GrailsMicronautBeanProcessor for current builder
138      * @deprecated please declare {@link MicronautBeanImporter} bean to avoid multiple Micronaut application context
139      * inside single application.
140      */
141     public GrailsMicronautBeanProcessor build() {
142         try {
143             throw new IllegalStateException();
144         } catch (IllegalStateException th) {
145             GrailsMicronautBeanProcessor.LOGGER.error("Old style of importing Micronaut beans used. This will lead to having multiple Micronaut application context in the application");
146         }
147         return new GrailsMicronautBeanProcessor(getMicronautBeanQualifiers(), getCustomizers(), getExpectedMapProperties());
148     }
149 }
