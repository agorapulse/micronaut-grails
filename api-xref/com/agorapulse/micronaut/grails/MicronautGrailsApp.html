74      private class MicronautGrailsAppContextConfiguration implements ApplicationContextConfiguration {
75          private final ClassLoader applicationClassLoader;
76          private final MicronautGrailsAutoConfiguration configuration;
77  
78          public MicronautGrailsAppContextConfiguration(
79              ClassLoader applicationClassLoader,
80              MicronautGrailsAutoConfiguration configuration
81          ) {
82              this.applicationClassLoader = applicationClassLoader;
83              this.configuration = configuration;
84          }
85  
86          @Override
87          @NonNull
88          public List<String> getEnvironments() {
89              List<String> environments = new ArrayList<>();
90              environments.add(ENVIRONMENT);
91              environments.add(configuration.getCompatibilityMode().getEnvironment());
92              if (getConfiguredEnvironment() != null) {
93                  environments.addAll(Arrays.asList(getConfiguredEnvironment().getActiveProfiles()));
94              }
95              return environments;
96          }
97  
98          @Override
99          public Optional<Boolean> getDeduceEnvironments() {
100             return Optional.of(false);
101         }
102 
103         @Override @NonNull
104         public ClassLoader getClassLoader() {
105             return applicationClassLoader;
106         }
107 
108         public MicronautGrailsAutoConfiguration getConfiguration() {
109             return configuration;
110         }
111     }

163     /**
164      * Strategy method used to create the {@link org.springframework.context.ApplicationContext}. By default this
165      * method will respect any explicitly set application context or application context
166      * class before falling back to a suitable default.
167      *
168      * @return the application context (not yet refreshed)
169      * @see #setApplicationContextClass(Class)
170      */
171     protected ConfigurableApplicationContext createSpringApplicationContext() {
172         Class<?> contextClass;
173         try {
174             switch (this.getWebApplicationType()) {
175                 case SERVLET:
176                     contextClass = Class.forName(DEFAULT_SERVLET_WEB_CONTEXT_CLASS);
177                     break;
178                 case REACTIVE:
179                     contextClass = Class.forName(DEFAULT_REACTIVE_WEB_CONTEXT_CLASS);
180                     break;
181                 default:
182                     contextClass = Class.forName(DEFAULT_CONTEXT_CLASS);
183             }
184         } catch (ClassNotFoundException ex) {
185             throw new IllegalStateException("Unable create a default ApplicationContext, please specify an ApplicationContextClass", ex);
186         }
187         return (ConfigurableApplicationContext) BeanUtils.instantiateClass(contextClass);
188     }

199     @Override
200     protected void configureEnvironment(ConfigurableEnvironment environment, String[] args) {
201         super.configureEnvironment(environment, args);
202         environment.addActiveProfile(ENVIRONMENT);
203 
204         MicronautGrailsAutoConfiguration app = getApplication();
205         environment.addActiveProfile(app.getCompatibilityMode().getEnvironment());
206     }

208     @Override
209     protected ConfigurableApplicationContext createApplicationContext() {
210         setAllowBeanDefinitionOverriding(true);
211 
212         ConfigurableApplicationContext applicationContext = createSpringApplicationContext();
213 
214         long now = System.currentTimeMillis();
215 
216         MicronautGrailsAutoConfiguration app = getApplication();
217         ClassLoader applicationClassLoader = GrailsApp.class.getClassLoader();
218         MicronautGrailsAppContextConfiguration micronautConfiguration = new MicronautGrailsAppContextConfiguration(applicationClassLoader, app);
219 
220         List<Class<?>> beanExcludes = new ArrayList<>();
221         beanExcludes.add(ConversionService.class);
222         beanExcludes.add(org.springframework.core.env.Environment.class);
223         beanExcludes.add(PropertyResolver.class);
224         beanExcludes.add(ConfigurableEnvironment.class);
225         ClassUtils.forName("com.fasterxml.jackson.databind.ObjectMapper", getClassLoader()).ifPresent(beanExcludes::add);
226         ApplicationContext micronautContext = new MicronautGrailsAppContext(micronautConfiguration);
227 
228         micronautContext.getEnvironment().addPropertySource("grails-config", Collections.singletonMap(MicronautBeanFactoryConfiguration.PREFIX + ".bean-excludes", beanExcludes));
229         micronautContext.registerSingleton(MicronautGrailsAutoConfiguration.class, app);
230         micronautContext.start();
231 
232         ConfigurableApplicationContext parentContext = micronautContext.getBean(ConfigurableApplicationContext.class);
233         applicationContext.setParent(parentContext);
234         applicationContext.addApplicationListener(new MicronautShutdownListener(micronautContext));
235 
236         LOGGER.info("Started Micronaut Parent Application Context in " + (System.currentTimeMillis() - now) + " ms");
237 
238         if (isEnableBeanCreationProfiler()) {
239             BeanCreationProfilingPostProcessor processor = new BeanCreationProfilingPostProcessor();
240             applicationContext.getBeanFactory().addBeanPostProcessor(processor);
241             applicationContext.addApplicationListener(processor);
242         }
243         return applicationContext;
244     }

246     @NonNull
247     private MicronautGrailsAutoConfiguration getApplication() {
248         try {
249             return getAllSources()
250                 .stream()
251                 .filter(s -> s instanceof Class<?> && MicronautGrailsAutoConfiguration.class.isAssignableFrom((Class<?>)s))
252                 .findFirst()
253                 .map(s -> BeanUtils.instantiateClass((Class<?>) s, MicronautGrailsAutoConfiguration.class))
254                 .orElseGet(MicronautGrailsAutoConfiguration::new);
255         } catch (BeanInstantiationException e) {
256             LOGGER.error("Failed to instantiate class" + getMainApplicationClass(), e);
257             return new MicronautGrailsAutoConfiguration();
258         }
259     }
