18  package com.agorapulse.micronaut.grails;
19  
20  import io.micronaut.context.Qualifier;
21  import io.micronaut.inject.BeanDefinition;
22  import org.slf4j.Logger;
23  import org.slf4j.LoggerFactory;
24  import org.springframework.beans.BeansException;
25  import org.springframework.beans.factory.DisposableBean;
26  import org.springframework.beans.factory.config.BeanFactoryPostProcessor;
27  import org.springframework.beans.factory.config.ConfigurableListableBeanFactory;
28  import org.springframework.beans.factory.support.BeanDefinitionBuilder;
29  import org.springframework.beans.factory.support.DefaultListableBeanFactory;
30  import org.springframework.context.ApplicationContext;
31  import org.springframework.context.ApplicationContextAware;
32  
33  import io.micronaut.core.annotation.NonNull;
34  import java.net.URLClassLoader;
35  import java.util.Arrays;
36  import java.util.Collection;
37  import java.util.Map;
38  import java.util.Optional;
39  import java.util.stream.Collectors;
40  
41  /**
42   * Adds Micronaut beans to a Grails' Spring application context.  This processor will
43   * find all of the Micronaut beans of the specified types and add them as beans to the Spring application context.
44   *
45   * No property translation will happen.
46   * <p>
47   *
48   * @since 2.0.3
49   */
50  public class DefaultGrailsMicronautBeanProcessor implements BeanFactoryPostProcessor, DisposableBean, ApplicationContextAware {
51  
52      static final Logger LOGGER = LoggerFactory.getLogger(DefaultGrailsMicronautBeanProcessor.class);
53  
54      private static final String MICRONAUT_BEAN_TYPE_PROPERTY_NAME = "micronautBeanType";
55      private static final String MICRONAUT_CONTEXT_PROPERTY_NAME = "micronautContext";
56      private static final String MICRONAUT_QUALIFIER_PROPERTY_NAME = "micronautQualifier";
57      private static final String MICRONAUT_SINGLETON_PROPERTY_NAME = "micronautSingleton";
58  
59      private io.micronaut.context.ApplicationContext micronautContext;
60      private ApplicationContext springContext;
61      private final Map<String, TypeAndQualifier<?>> micronautBeanQualifiers;
62  
63      /**
64       * @param qualifiers the names and qualifiers of the Micronaut beans which should be added to the
65       *                   Spring application context.
66       */
67      DefaultGrailsMicronautBeanProcessor(Map<String, TypeAndQualifier<?>> qualifiers) {
68          this.micronautBeanQualifiers = qualifiers;
69      }
70  
71      @Override
72      public void postProcessBeanFactory(ConfigurableListableBeanFactory beanFactory) throws BeansException {
73          micronautContext = initializeMicronautContext();
74  
75          NoClassDefFoundError noClassDefFoundError = null;
76  
77          for (Map.Entry<String, TypeAndQualifier<?>> entry : micronautBeanQualifiers.entrySet()) {
78              String name = entry.getKey();
79              Class type = entry.getValue().getType();
80              Qualifier micronautBeanQualifier = entry.getValue().getQualifier();
81              try {
82                  Collection<BeanDefinition<?>> beanDefinitions = type == null
83                      ? micronautContext.getBeanDefinitions(micronautBeanQualifier)
84                      : micronautContext.getBeanDefinitions(type, micronautBeanQualifier);
85  
86                  if (beanDefinitions.size() > 1) {
87                      throw new IllegalArgumentException("There is too many candidates of type '" + type + "' for qualifier '" + micronautBeanQualifier + "'! Candidates: " + beanDefinitions);
88                  }
89  
90                  Optional<BeanDefinition<?>> firstBean = beanDefinitions.stream().findFirst();
91                  BeanDefinition<?> definition = firstBean.orElseThrow(()-> {
92                          String message = "There is no candidate for type " + type + " and qualifier " + micronautBeanQualifier + "\n"
93                              + "Known beans:\n" + micronautContext.getAllBeanDefinitions().stream().map(d -> d.getBeanType().getName() + " " + d.getName()).collect(Collectors.joining("\n  "));
94  
95                          return new IllegalArgumentException(message);
96                  });
97  
98                  final BeanDefinitionBuilder beanDefinitionBuilder = BeanDefinitionBuilder
99                      .rootBeanDefinition(GrailsMicronautBeanFactory.class);
100                 beanDefinitionBuilder.addPropertyValue(MICRONAUT_BEAN_TYPE_PROPERTY_NAME, type == null ? definition.getBeanType() : type);
101                 beanDefinitionBuilder.addPropertyValue(MICRONAUT_QUALIFIER_PROPERTY_NAME, micronautBeanQualifier);
102                 beanDefinitionBuilder.addPropertyValue(MICRONAUT_CONTEXT_PROPERTY_NAME, micronautContext);
103                 beanDefinitionBuilder.addPropertyValue(MICRONAUT_SINGLETON_PROPERTY_NAME, definition.isSingleton());
104 
105                 ((DefaultListableBeanFactory) beanFactory).registerBeanDefinition(name, beanDefinitionBuilder.getBeanDefinition());
106             } catch (NoClassDefFoundError error) {
107                 LOGGER.error("Exception loading class for qualifier {}. Bean {} will not be available in the runtime", micronautBeanQualifier, name);
108                 LOGGER.error("Current class loader: {}", printClassLoader(getClass().getClassLoader()));
109                 LOGGER.error("Parent class loader: {}",  printClassLoader(getClass().getClassLoader().getParent()));
110                 LOGGER.error("Current class path: {}", System.getProperty("java.class.path"));
111                 noClassDefFoundError = error;
112             }
113         }
114 
115         if (noClassDefFoundError == null) {
116             if (LOGGER.isInfoEnabled()) {
117                 LOGGER.info("Successfully added following beans to the spring contest {} ", micronautBeanQualifiers);
118                 LOGGER.debug("Current class loader: {}", printClassLoader(getClass().getClassLoader()));
119                 LOGGER.debug("Parent class loader: {}",  printClassLoader(getClass().getClassLoader().getParent()));
120                 LOGGER.trace("Current class path: {}", System.getProperty("java.class.path"));
121             }
122             return;
123         }
124 
125         throw noClassDefFoundError;
126     }
127 
128     protected io.micronaut.context.ApplicationContext initializeMicronautContext() {
129         return springContext.getBean(MicronautContextHolder.class).getContext();
130     }
131 
132     private static String printClassLoader(ClassLoader classLoader) {
133         if (classLoader instanceof URLClassLoader) {
134             return "URLClassLoader for URLS:" + Arrays.toString(((URLClassLoader) classLoader).getURLs());
135         }
136         if (classLoader == null) {
137             return null;
138         }
139         return classLoader.toString();
140     }
141 
142     @Override
143     public void destroy() {
144         if (micronautContext != null) {
145             micronautContext.close();
146         }
147     }
148 
149     @Override
150     public void setApplicationContext(@NonNull ApplicationContext applicationContext) throws BeansException {
151         this.springContext = applicationContext;
152     }
153 }
