44  /**
45   * Adds Micronaut beans to a Grails' Spring application context.  This processor will
46   * find all of the Micronaut beans of the specified types
47   * and add them as beans to the Spring application context.
48   * <p>
49   * The Grails properties can be translated to Micronaut properties if required.
50   *
51   * @since 1.0
52   */
53  public class GrailsMicronautBeanProcessor implements BeanFactoryPostProcessor, DisposableBean, EnvironmentAware, ApplicationContextAware {
54  
55      static final Logger LOGGER = LoggerFactory.getLogger(GrailsMicronautBeanProcessor.class);
56  
57      /**
58       * @deprecated please declare {@link MicronautBeanImporter} bean to avoid multiple Micronaut application context
59       * inside single application.
60       */
61      public static class Builder extends MicronautBeanImporter {
62  
63          Builder(PropertyTranslatingCustomizer customizer) {
64              customize(customizer);
65          }
66  
67      }
68  
69      /**
70       * Starts creation of bean processor using default {@link PropertyTranslatingCustomizer#grails()} customizer.
71       * @return bean processor builder using {@link PropertyTranslatingCustomizer#grails()}
72       * @deprecated please declare {@link MicronautBeanImporter} bean instead to avoid multiple Micronaut application context
73       * inside single application.
74       */
75      public static GrailsMicronautBeanProcessor.Builder builder() {
76          return builder(grails());
77      }
78  
79      /**
80       * Starts creation of bean processor using given customizer.
81       * @param customizerBuilder customizer being used
82       * @return bean processor builder using given cusotomizer
83       * @deprecated please declare {@link MicronautBeanImporter} bean instead to avoid multiple Micronaut application context
84       * inside single application.
85       */
86      public static GrailsMicronautBeanProcessor.Builder builder(PropertyTranslatingCustomizer.Builder customizerBuilder) {
87          return builder(customizerBuilder.build());
88      }
89  
90      /**
91       * Starts creation of bean processor using given customizer.
92       * @param customizer customizer being used
93       * @return bean processor builder using given cusotomizer
94       * @deprecated please declare {@link MicronautBeanImporter} bean instead to avoid multiple Micronaut application context
95       * inside single application.
96       */
97      public static GrailsMicronautBeanProcessor.Builder builder(PropertyTranslatingCustomizer customizer) {
98          return new Builder(customizer);
99      }

101     private static final String MICRONAUT_BEAN_TYPE_PROPERTY_NAME = "micronautBeanType";
102     private static final String MICRONAUT_CONTEXT_PROPERTY_NAME = "micronautContext";
103     private static final String MICRONAUT_QUALIFIER_PROPERTY_NAME = "micronautQualifier";
104     private static final String MICRONAUT_SINGLETON_PROPERTY_NAME = "micronautSingleton";
105 
106     private io.micronaut.context.ApplicationContext micronautContext;
107     private ApplicationContext springContext;
108     private final Map<String, TypeAndQualifier<?>> micronautBeanQualifiers;
109     private final List<PropertyTranslatingCustomizer> customizers;
110     private final List<String> expectedMapProperties;
111     private Environment environment;
112 
113     /**
114      * @param qualifiers the names and qualifiers of the Micronaut beans which should be added to the
115      *                   Spring application context.
116      * @param customizers properties translation customizer
117      * @param expectedMapProperties list of properties' prefixes which should be converted to map
118      */
119     GrailsMicronautBeanProcessor(Map<String, TypeAndQualifier<?>> qualifiers, List<PropertyTranslatingCustomizer> customizers, List<String> expectedMapProperties) {
120         this.customizers = customizers;
121         this.micronautBeanQualifiers = qualifiers;
122         this.expectedMapProperties = expectedMapProperties;
123     }
124 
125     @Override
126     public void postProcessBeanFactory(ConfigurableListableBeanFactory beanFactory) throws BeansException {
127         if (environment == null) {
128             throw new IllegalStateException("Spring environment not set!");
129         }
130 
131         MicronautGrailsApp.Configuration configuration = null;
132         Set<String> packages = new LinkedHashSet<>();
133 
134         try {
135             MicronautContextHolder holder = springContext.getBean(MicronautContextHolder.class);
136             io.micronaut.context.ApplicationContext context = holder.getContext();
137             packages.addAll(context.getEnvironment().getPackages());
138             if (
139                     context.getEnvironment().getActiveNames().contains(MicronautGrailsApp.ENVIRONMENT)
140                 &&
141                     !context.getEnvironment().getActiveNames().contains(MicronautGrailsApp.ENVIRONMENT_LEGACY)
142             ) {
143                 LOGGER.info("Reusing existing application context. Property customizations and exclusions are disabled.");
144                 micronautContext = context;
145             } else {
146                 LOGGER.info("Running in the bridge or the legacy mode - two Micronaut context will be present");
147             }
148             configuration = context.findBean(MicronautGrailsApp.Configuration.class).orElse(null);
149         } catch (NoSuchBeanDefinitionException ignored) {
150             LOGGER.info("Running using GrailsApp instead of MicronautGrailsApp - two Micronaut context will be present");
151         }
152 
153         if (micronautContext == null) {
154             micronautContext = new GrailsPropertyTranslatingApplicationContext(environment, of(collapse(customizers)), expectedMapProperties);
155 
156             packages.forEach(micronautContext.getEnvironment()::addPackage);
157 
158             if (configuration != null) {
159                 configuration.applyToEnvironment(micronautContext.getEnvironment());
160             }
161 
162             micronautContext.start();
163         }
164 
165         if (micronautContext.getEnvironment().getActiveNames().contains(MicronautGrailsApp.ENVIRONMENT_STRICT)) {
166             LOGGER.info("Running in the strict mode. You must use @Inject for all Micronaut beans. Property customizations and exclusiong are disabled.");
167             return;
168         }
169 
170         NoClassDefFoundError noClassDefFoundError = null;
171 
172         for (Map.Entry<String, TypeAndQualifier<?>> entry : micronautBeanQualifiers.entrySet()) {
173             String name = entry.getKey();
174             Class type = entry.getValue().getType();
175             Qualifier micronautBeanQualifier = entry.getValue().getQualifier();
176             try {
177                 Collection<BeanDefinition<?>> beanDefinitions = type == null
178                     ? micronautContext.getBeanDefinitions(micronautBeanQualifier)
179                     : micronautContext.getBeanDefinitions(type, micronautBeanQualifier);
180 
181                 if (beanDefinitions.size() > 1) {
182                     throw new IllegalArgumentException("There is too many candidates of type " + type + " for " + micronautBeanQualifier + "! Candidates: " + beanDefinitions);
183                 }
184 
185                 Optional<BeanDefinition<?>> firstBean = beanDefinitions.stream().findFirst();
186                 BeanDefinition<?> definition = firstBean.orElseThrow(()-> new IllegalArgumentException("There is no candidate for " + micronautBeanQualifier));
187 
188                 final BeanDefinitionBuilder beanDefinitionBuilder = BeanDefinitionBuilder
189                     .rootBeanDefinition(GrailsMicronautBeanFactory.class);
190                 beanDefinitionBuilder.addPropertyValue(MICRONAUT_BEAN_TYPE_PROPERTY_NAME, type == null ? definition.getBeanType() : type);
191                 beanDefinitionBuilder.addPropertyValue(MICRONAUT_QUALIFIER_PROPERTY_NAME, micronautBeanQualifier);
192                 beanDefinitionBuilder.addPropertyValue(MICRONAUT_CONTEXT_PROPERTY_NAME, micronautContext);
193                 beanDefinitionBuilder.addPropertyValue(MICRONAUT_SINGLETON_PROPERTY_NAME, definition.isSingleton());
194 
195                 ((DefaultListableBeanFactory) beanFactory).registerBeanDefinition(name, beanDefinitionBuilder.getBeanDefinition());
196             } catch (NoClassDefFoundError error) {
197                 LOGGER.error("Exception loading class for qualifier {}. Bean {} will not be available in the runtime", micronautBeanQualifier, name);
198                 LOGGER.error("Current class loader: {}", printClassLoader(getClass().getClassLoader()));
199                 LOGGER.error("Parent class loader: {}",  printClassLoader(getClass().getClassLoader().getParent()));
200                 LOGGER.error("Current class path: {}", System.getProperty("java.class.path"));
201                 noClassDefFoundError = error;
202             }
203         }
204 
205         if (noClassDefFoundError == null) {
206             if (LOGGER.isInfoEnabled()) {
207                 LOGGER.info("Successfully added following beans to the spring contest {} ", micronautBeanQualifiers);
208                 LOGGER.info("Current class loader: {}", printClassLoader(getClass().getClassLoader()));
209                 LOGGER.info("Parent class loader: {}",  printClassLoader(getClass().getClassLoader().getParent()));
210                 LOGGER.info("Current class path: {}", System.getProperty("java.class.path"));
211             }
212             return;
213         }
214 
215         throw noClassDefFoundError;
216     }
217 
218     private static String printClassLoader(ClassLoader classLoader) {
219         if (classLoader instanceof URLClassLoader) {
220             return "URLClassLoader for URLS:" + Arrays.toString(((URLClassLoader) classLoader).getURLs());
221         }
222         if (classLoader == null) {
223             return null;
224         }
225         return classLoader.toString();
226     }
227 
228     @Override
229     public void destroy() {
230         if (micronautContext != null) {
231             micronautContext.close();
232         }
233     }

237         this.environment = environment;
238     }

241     public void setApplicationContext(@Nonnull ApplicationContext applicationContext) throws BeansException {
242         this.springContext = applicationContext;
243     }
