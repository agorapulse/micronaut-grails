18  package com.agorapulse.micronaut.grails;
19  
20  public enum CompatibilityMode {
21      /**
22       * Runs the application in the legacy mode.
23       *
24       * <ul>
25       *     <li>Two Micronaut contexts are created - one by Grails App and another one by Micronaut Grails for beans imported using {@link MicronautBeanImporter}</li>
26       *     <li>Micronaut beans declared using {@link MicronautBeanImporter} can be injected by name without <code>@Inject</code> annotation</li>
27       *     <li>Property prefixes are stripped by beans imported using {@link MicronautBeanImporter} according to existing {@link PropertyTranslatingCustomizer} beans</li>
28       *     <li>Properties are ignored by beans imported using {@link MicronautBeanImporter} according to existing {@link PropertyTranslatingCustomizer} beans</li>
29       * </ul>
30       */
31      @Deprecated
32      LEGACY(MicronautGrailsApp.ENVIRONMENT_LEGACY),
33  
34      /**
35       * Runs the application in the bridge mode.
36       *
37       * <ul>
38       *     <li>Only one Micronaut context is created - the one by Grails App</li>
39       *     <li>Micronaut beans declared using {@link MicronautBeanImporter} can be injected by name without <code>@Inject</code> annotation</li>
40       *     <li>Property prefixes are <em>no longer stripped</em> by beans imported using {@link MicronautBeanImporter} according to existing {@link PropertyTranslatingCustomizer} beans</li>
41       *     <li>Properties are <em>no longer ignored</em> by beans imported using {@link MicronautBeanImporter} according to existing {@link PropertyTranslatingCustomizer} beans</li>
42       * </ul>
43       */
44      BRIDGE(MicronautGrailsApp.ENVIRONMENT_BRIDGE),
45  
46      /**
47       * Runs the application in the bridge mode.
48       *
49       * <ul>
50       *     <li>Only one Micronaut context is created - the one by Grails App</li>
51       *     <li>All Micronaut beans must use <code>@Inject</code> annotation to be injected</li>
52       *     <li>Property prefixes are <em>no longer stripped</em> by beans imported using {@link MicronautBeanImporter} according to existing {@link PropertyTranslatingCustomizer} beans</li>
53       *     <li>Properties are <em>no longer ignored</em> by beans imported using {@link MicronautBeanImporter} according to existing {@link PropertyTranslatingCustomizer} beans</li>
54       * </ul>
55       */
56      STRICT(MicronautGrailsApp.ENVIRONMENT_STRICT);
57  
58      private final String environment;
59  
60      CompatibilityMode(String environment) {
61          this.environment = environment;
62      }
63  
64      public String getEnvironment() {
65          return environment;
66      }
67  
68  }
